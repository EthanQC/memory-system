## 2. list smoke

## 3. show smoke

uv run memoryd show "$SLUG"

## 4. delete sandbox 安全测

# 用 isolated MEMORYD_DATA_ROOT，不污染真数据
SANDBOX=/tmp/memoryd-delete-test
rm -rf $SANDBOX && mkdir -p $SANDBOX/scopes/h_test/sessions

cat > $SANDBOX/scopes/h_test/sessions/test-delete.md <<'EOF'
---
title: test
slug: test-delete
scope_hash: h_test
type: session
triggers: [test]
source: manual
created_at: 2026-05-16T00:00:00+00:00
---
this is sandbox content soon to be deleted
EOF

MEMORYD_DATA_ROOT=$SANDBOX uv run memoryd show test-delete
MEMORYD_DATA_ROOT=$SANDBOX uv run memoryd delete test-delete --force
test ! -f $SANDBOX/scopes/h_test/sessions/test-delete.md && echo "OK deleted"
rm -rf $SANDBOX

## 5. promote（需先有 pending promotion）

# Pending promotion 在 ANTHROPIC_API_KEY 配置后跑 analyze-session 才生成。
# 假设有的话，拿 promotion_id：
PID=$(uv run memoryd digest --json 2>/dev/null | jq -r '.promotions[]? | select(.status=="pending") | .id' | head -1)
if [ -n "$PID" ]; then
  uv run memoryd promote "$PID"
  # 应在 scopes/<hash>/<type>s/promoted-<id>-<slug>.md 出现新 .md
fi
